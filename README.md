public String build() {
  StringBuffer phrase  = new StringBuffer();
  List<String> words = ImmutableList.of("1", "Clean", "24", "7", "code", "14", " ");

  words.stream()
       .filter(this::isValidString)
       .forEach(phrase::append);

  return phrase.toString();
}
